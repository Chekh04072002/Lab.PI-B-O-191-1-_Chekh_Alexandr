```c++
#include <iostream>
#include <cmath>
bool isEqual(double a, double b) {
	double e = 0.00000001;
	return (abs(a - b) < e); 
}
double a, b;
double f(double x) {
	if (x < a || isEqual(x, a)) {
		return log(x) * sin(x);
	}
	else if (x > a && x < b) {
		return (x) * (x)* cos(x);
	}
	else if (x >= b) {
		return (x) * (x) * (x) * (x) * (x);
	}

}
int main() {
	double xStart, xEnd, dx;
	setlocale(LC_ALL, "russian");
	std::cout << "Введите A, B, xнач, xкон, dx: ";
	std::cin >> a >> b >> xStart >> xEnd >> dx;

	double max = -100000000, min = 100000000;
	for (double x = xStart; x <= xEnd || isEqual(x, xEnd) ; x = x + dx) {
		double val = f(x);
		if (val < min) min = val;
		if (val > max) max = val;

		std::cout << "f(" << x << ") = " << val << std::endl;
	}

	std::cout << "Min: " << min << std::endl << "Max: " << max << std::endl;

	return 0;
}
```
